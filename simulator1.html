async function drawAll() { ctx.clearRect(0, 0, canvas.width, canvas.height); const parts = ['main', '1', '2', '3', '4', '5']; for (const part of parts) { const fabricUrl = currentFabrics[part]; const maskUrl = part === 'main' ? 'main_front.PNG' : part + '_front.PNG'; if (fabricUrl) { const fabricImg = await loadImage(fabricUrl); const maskImg = await loadImage(maskUrl); if (fabricImg && maskImg) { // --- ここから魔法の書き換え --- // 1. 別のキャンバスに「型紙」を描いて、白を透明に変換する処理 const tempCanvas = document.createElement('canvas'); tempCanvas.width = canvas.width; tempCanvas.height = canvas.height; const tCtx = tempCanvas.getContext('2d'); tCtx.drawImage(maskImg, 0, 0, canvas.width, canvas.height); // 2. 生地を描く ctx.save(); ctx.drawImage(fabricImg, 0, 0, canvas.width, canvas.height); // 3. 型抜きを実行 ctx.globalCompositeOperation = 'destination-in'; ctx.drawImage(tempCanvas, 0, 0); ctx.restore(); } } } // 枠線の描画（透過してなくても見えるように少し工夫） const outlines = ['main_front.PNG', '1_front.PNG', '2_front.PNG', '3_front.PNG', '4_front.PNG', '5_front.PNG', '6_front.PNG']; for (const url of outlines) { const img = await loadImage(url); if (img) { ctx.save(); ctx.globalCompositeOperation = 'multiply'; // 重ね塗りモード（白を透明化） ctx.drawImage(img, 0, 0, canvas.width, canvas.height); ctx.restore(); } } }
